CREATE TABLE Students( id integer primary key, first_name text, last_name text, remote bit, relocate bit, linkedin text, github text, twitter text, username text, password text)

CREATE TABLE Employer( id integer primary key, company_name text, remote bit, email text, username text, password text)

#password security decision

CREATE TABLE Student_Phone(id integer primary key, studentID integer, number text, FOREIGN KEY(studentID) REFERENCES Students(id))

CREATE TABLE Employer_Phone(id integer primary key, employerID integer, number text, FOREIGN KEY(employerID) REFERENCES Employer(id))

CREATE TABLE Student_Favorites(id integer primary key,studentID integer, employerID integer, FOREIGN KEY(studentID) REFERENCES Students(id),
FOREIGN KEY(employerID) REFERENCES Employer(id))

CREATE TABLE Employer_Favorites(id integer primary key, employerID integer, studentID integer, FOREIGN KEY(studentID) REFERENCES Students(id),
FOREIGN KEY(employerID) REFERENCES Employer(id))
