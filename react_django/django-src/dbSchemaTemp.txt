CREATE TABLE Students( id integer primary key, first_name text, last_name text, remote bit, relocate bit, linkedin text, github text, twitter text, city text, state text, username text, password text)

CREATE TABLE Employers( id integer primary key, company_name text, currently_subscribed bit,  email text, city text, state text, username text, password text)

#password security decision

CREATE TABLE Student_Phones(id integer primary key, studentID integer, number text, FOREIGN KEY(studentID) REFERENCES Students(id))

CREATE TABLE Employer_Phones(id integer primary key, employerID integer, number text, FOREIGN KEY(employerID) REFERENCES Employers(id))

CREATE TABLE Student_Favorites(id integer primary key,studentID integer, employerID integer, FOREIGN KEY(studentID) REFERENCES Students(id),
FOREIGN KEY(employerID) REFERENCES Employers(id))

CREATE TABLE Employer_Favorites(id integer primary key, employerID integer, studentID integer, FOREIGN KEY(studentID) REFERENCES Students(id),
FOREIGN KEY(employerID) REFERENCES Employers(id))

CREATE TABLE Billing(id integer primary key, CC_Number text, Expiration text, CVV text, employerID integer, FOREIGN KEY(employerID) REFERENCES Employers(id))

CREATE TABLE Messages(id integer primary key, content text, post_time text, studentID integer, employerID integer, FOREIGN KEY(employerID) REFERENCES Employers(id), FOREIGN KEY(studentID) REFERENCES Student(id))
